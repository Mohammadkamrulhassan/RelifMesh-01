\section{Development Environment}
\textbf{Prerequisites:} Node.js $\ge$ 20.x LTS, npm $\ge$ 10.x, MongoDB $\ge$ 8.x

\textbf{Setup:}
\begin{lstlisting}[language=bash]
git clone https://github.com/Team-Skipper/reliefmesh.git
cd backend && cp ../.env.example .env && npm install && npm run seed && npm run dev
cd ../frontend && npm install && npm run dev
\end{lstlisting}

\section{Coding Standards}
\begin{itemize}[nosep]
    \item Language: JavaScript (ES2022+), no TypeScript (team familiarity)
    \item Linter: ESLint with Airbnb config
    \item Formatter: Prettier (2-space indent, single quotes)
    \item Files: kebab-case, Functions: camelCase, Components: PascalCase, DB: snake\_case
    \item All async operations use \texttt{async/await}
\end{itemize}
