\section{Design System}

\subsection{Color Palette (Result09 Design System)}
\begin{itemize}[nosep]
    \item \textbf{Light Theme:} Primary \#2563eb, Surface \#ffffff, Text \#1e293b, Danger \#ef4444, Warning \#f59e0b, Success \#22c55e
    \item \textbf{Dark Theme:} Surface \#0f172a, Card \#1e293b, Text \#f1f5f9
    \item \textbf{Sidebar:} Dark panel with \#0f172a background, \#3b82f6 active indicator
\end{itemize}

\subsection{Typography}
Inter font family (300--900 weights), App Title 22px/700, Section Header 18px/600, Body 16px/400, Caption 13px/400

\subsection{Component Standards}
\begin{itemize}[nosep]
    \item Buttons: min height 40px, border-radius 8px, 6 variants
    \item Input fields: min height 44px, clear label, error/hint/success states
    \item Cards: 12px border-radius, 1px border, subtle shadow
    \item Data tables: full-width, striped rows, sticky header
    \item Sidebar: fixed dark panel, collapsible, mobile overlay
    \item Topbar: breadcrumbs, theme toggle, sync status indicator
\end{itemize}
