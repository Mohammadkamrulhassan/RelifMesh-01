\section{Git Discipline}
\begin{itemize}[nosep]
    \item No direct commits to \texttt{main}
    \item All features via pull requests with $\ge$ 1 reviewer
    \item Meaningful commit messages (\texttt{feat:}, \texttt{fix:}, \texttt{docs:}, etc.)
    \item Branches: \texttt{main} $\rightarrow$ \texttt{develop} $\rightarrow$ \texttt{feature/xxx}
\end{itemize}
