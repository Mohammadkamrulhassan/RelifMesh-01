\section{Test Plan}
\textbf{Objective:} Verify all functional requirements (FR-01 to FR-44), validate NFRs, ensure RBAC enforcement, confirm offline sync and duplicate detection.

\textbf{Scope:} Unit testing, Integration testing, System testing, User Acceptance Testing (UAT).

\textbf{Tools:} Jest, Supertest, React Testing Library, Postman, Chrome DevTools, Lighthouse.

\section{Unit Test Cases (29 Passing)}
\begin{itemize}[nosep]
    \item \textbf{Authentication (TC-01 to TC-04):} Valid login returns JWT $\checkmark$, Invalid password rejected $\checkmark$, Non-existent user rejected $\checkmark$, Role embedded in JWT $\checkmark$
    \item \textbf{Household (TC-05 to TC-08):} Valid registration saves HH-ID $\checkmark$, Duplicate NID rejected $\checkmark$, Missing field fails validation $\checkmark$, GPS stored correctly $\checkmark$
    \item \textbf{Distribution (TC-09 to TC-11):} Valid log saved $\checkmark$, Unknown HH-ID rejected $\checkmark$, Quantity must be positive $\checkmark$
    \item \textbf{Duplicate Detection (TC-12 to TC-16):} Same item within 7 days $\rightarrow$ duplicate $\checkmark$, Same item after 8 days $\rightarrow$ no duplicate $\square$, Different item $\rightarrow$ no duplicate $\checkmark$, Override accepted with reason $\checkmark$, Override rejected without reason $\checkmark$
\end{itemize}
