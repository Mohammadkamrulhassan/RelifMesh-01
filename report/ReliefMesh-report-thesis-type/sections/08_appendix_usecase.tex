% Define a reusable stick-figure actor
\newcommand{\actor}[4]{%
  \node[circle, draw, thick, minimum size=0.45cm, inner sep=0pt] (#1_head) at (#2, #3+0.9) {};
  \draw[thick] (#2, #3+0.675) -- (#2, #3+0.35);
  \draw[thick] (#2-0.35, #3+0.55) -- (#2+0.35, #3+0.55);
  \draw[thick] (#2, #3+0.35) -- (#2, #3);
  \draw[thick] (#2, #3) -- (#2-0.3, #3-0.35);
  \draw[thick] (#2, #3) -- (#2+0.3, #3-0.35);
  \node[below=4pt, font=\small\sffamily, align=center] at (#2, #3-0.35) {#4};
}
